    ```python
    def truncate_user_agent(ua_string):
        """DSGVO-compliant UA truncation to browser family + major version.
        Returns 'Browser/Version' (e.g. 'Chrome/120') or 'unknown' on no match.
        Order matters: more-specific browsers (Edge, OPR, CriOS) MUST appear
        before Chrome/Safari in the alternation because Chrome UAs contain
        'Safari/' and Edge UAs contain 'Chrome/'.
        AUDIT-03 — Phase 1, v2.0 milestone.
        """
        if not ua_string:
            return "unknown"
        match = re.search(
            r'(Edge|Edg|OPR|CriOS|FxiOS|SamsungBrowser|UCBrowser|MSIE|Trident|Chrome|Firefox|Safari|Opera)[\/ ](\d+)',
            ua_string
        )
        if not match:
            return "unknown"
        browser = match.group(1)
        if browser == "Edg":
            browser = "Edge"
        elif browser in ("Trident", "MSIE"):
            browser = "IE"
        return f"{browser}/{match.group(2)}"
    ```
